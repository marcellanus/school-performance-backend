from flask import Flask, render_template, request, jsonify
import sqlite3

app = Flask(__name__)
app.secret_key = 'supersecretkey'

DATABASE = 'database/school.db'

def get_db_connection():
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row
    return conn

@app.route('/dashboard')
def dashboard():
    conn = get_db_connection()
    classes = conn.execute('SELECT * FROM classes').fetchall()
    conn.close()
    return render_template('dashboard.html', classes=classes)

@app.route('/api/class_performance/<int:class_id>')
def class_performance(class_id):
    conn = get_db_connection()
    results = conn.execute('''
        SELECT subjects.name AS subject, 
               AVG((scores.assessment + scores.exam) / 2) AS average_score 
        FROM scores 
        JOIN subjects ON scores.subject_id = subjects.id 
        JOIN students ON scores.student_id = students.id 
        WHERE students.class_id = ? 
        GROUP BY subjects.name
    ''', (class_id,)).fetchall()
    conn.close()
    return jsonify([dict(row) for row in results])

@app.route('/api/recommendations/<int:class_id>')
def recommendations(class_id):
    conn = get_db_connection()
    results = conn.execute('''
        SELECT subjects.name AS subject, 
               AVG((scores.assessment + scores.exam) / 2) AS average_score 
        FROM scores 
        JOIN subjects ON scores.subject_id = subjects.id 
        JOIN students ON scores.student_id = students.id 
        WHERE students.class_id = ? 
        GROUP BY subjects.name
        HAVING average_score < 50
    ''', (class_id,)).fetchall()
    conn.close()
    return jsonify([dict(row) for row in results])

if __name__ == '__main__':
    app.run(debug=True)

# Login route modification to detect creator access
@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        email = request.form['email']
        password = request.form['password']
        conn = get_db_connection()
        user = conn.execute('SELECT * FROM users WHERE email = ? AND password = ?', (email, password)).fetchone()
        conn.close()
        if user:
            session['user_id'] = user['id']
            session['is_creator'] = (user['role'] == 'creator')
            flash('Welcome back!', 'success')
            return redirect(url_for('dashboard'))
        else:
            flash('Invalid credentials.', 'danger')
    return render_template('login.html')
